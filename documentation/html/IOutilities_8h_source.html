288 template<typename _DataType, typename _Scalar>
289 inline Eigen::Matrix<_Scalar, Eigen::Dynamic, 1> DataTypes::parseData(std::istream& in, long sizeInElements, bool shuffleBytes)
290 {
291  long sizeInBytes = sizeInElements * sizeof(_DataType);
292 
293  // parse data into temporary buffer
294  std::vector<_DataType> tempBuffer(sizeInElements);
295  in.read(reinterpret_cast<char*>(&tempBuffer[0]), sizeInBytes);
296  if (in.gcount() != sizeInBytes)
297  throw std::runtime_error("parseData: failed to read the expected amount of data.");
298 
299  // perform byte shuffling to correct endianness
300  if (shuffleBytes)
301  ByteOrder::shuffleBytes<_DataType>(&tempBuffer[0], sizeInElements);
302 
303  // create result vector
304  Eigen::Matrix<_Scalar, Eigen::Dynamic, 1> result(sizeInElements);
305  for (long i = 0; i < sizeInElements; ++i)
306  result(i) = static_cast<_Scalar>(tempBuffer[i]);
307 
308  return result;
309 }
